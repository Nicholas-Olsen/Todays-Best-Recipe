CREATE DATABASE IF NOT EXISTS RECIPE;
USE RECIPE;

CREATE TABLE recipes (
    rec_id INT AUTO_INCREMENT PRIMARY KEY,
    rec_name VARCHAR(255) NOT NULL,
    rec_descrip TEXT,
    rec_detail MEDIUMTEXT,
    rec_img VARCHAR(255),
    rec_type VARCHAR(45)
);

CREATE TABLE user_ingredients (
    user_id VARCHAR(30) NOT NULL,
    ing_name varchar(30) NOT NULL
);

CREATE TABLE users (
    user_id VARCHAR(30) NOT NULL PRIMARY KEY,
    nickname VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255)
);

CREATE TABLE users_list (
    user_id VARCHAR(30) NOT NULL PRIMARY KEY,
    list TEXT
);

CREATE TABLE recipe_steps (
    rec_id INT NOT NULL,                     -- 연결된 레시피 ID (recipes 테이블과 연결)
    step_number INT NOT NULL,                 -- 조리 순서 (기본 키 역할)
    step_description TEXT NOT NULL,           -- 조리 과정 설명
    
    PRIMARY KEY (rec_id, step_number),        -- 복합 기본 키 설정
    FOREIGN KEY (rec_id) REFERENCES recipes(rec_id) ON DELETE CASCADE
);
